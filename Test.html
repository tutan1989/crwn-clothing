public class GlobalVariables { public static String parameterValue; public static void setParameterValue(String value) { parameterValue = value; } public static String getParameterValue() { return parameterValue; } } public class Main { public static void main(String[] args) { if (args.length == 0) { System.out.println("No argument provided."); return; } String parameterValue = args[0]; // Set the global variable GlobalVariables.setParameterValue(parameterValue); // Create an instance of OtherClass OtherClass otherClass = new OtherClass(); otherClass.doSomethingWithParameter(); } } public class OtherClass { public void doSomethingWithParameter() { String parameterValue = GlobalVariables.getParameterValue(); System.out.println("Parameter value in OtherClass: " + parameterValue); // Use the parameterValue here } }
